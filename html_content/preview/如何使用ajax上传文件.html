使用ajax上传文件

使用ajax上传文件，需要将form表单中的数据序列化成一个FormData对象

  • 前端
      <form action="" id="form1">
      用户昵称：<input type="text" name="niceName"><br>
      用户头像：<input type="file" name="file"><br>
      <input type="button" value="上传" onclick="ajaxUpload()">
  </form>
  <script>
      function ajaxUpload(){
          var xhr=new XMLHttpRequest();
          xhr.open("post","./php/saveFiles.php");
          xhr.onload=function(){
              if(xhr.status==200){
                  console.log(xhr.responseText);
              }
          };
          //将一个form表单中的数据序列化成一个FormData对象
          var formData=new FormData(document.getElementById("form1"));
          xhr.send(formData);
      }
  </script>
